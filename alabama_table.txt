function groupsCurrentSortBySet() {

	var select = document.getElementById('tsort');
	var index = select.selectedIndex;
	var options = select.options;
	for (var n = 0; n &lt; options.length; n++) {
		var divId = options[n].value.replace(',', '_');
		if (n == index) {
			groupsSetStartEndValues(options[n].value.split(',')[0], divId);
			replaceClass(divId, 'displayoff', 'displayon');
			var elem = document.getElementById(divId + 'S');
			var text = elem == null ? 'n/a' : (elem.innerText || elem.textContent);
			if (text == 'text')
				text = 'n/a';
			document.getElementById('group1').value = text;
		}
		else
			replaceClass(divId, 'displayon', 'displayoff');
	}
}

// set start/end values for current sort by measure using current totals/zeros settings
function groupsSetStartEndValues(mcode, divId) {

	// no start/end sections for byvariables or text variables
	if (document.getElementById(divId + 'S') == null)
		return;

	var max = 0;
	var min = 0;
	var totals = document.form.O_show_totals.checked;
	var zeros = document.form.O_show_zeros.checked;
	if (totals &amp;&amp; zeros) {
		max = measures[mcode].max;
		min = measures[mcode].min;
	}
	else if (totals &amp;&amp; !zeros) {
		max = measures[mcode].maxWtNz;
		min = measures[mcode].minWtNz;
	}
	else if (!totals &amp;&amp; zeros) {
		max = measures[mcode].maxNtWz;
		min = measures[mcode].minNtWz;
	}
	else {
		max = measures[mcode].maxNtNz;
		min = measures[mcode].minNtNz;
	}

	var asc = divId.indexOf('ASCEND') &gt; -1;
	var elem = document.getElementById(divId + 'S');
	if (elem.innerText) 
		elem.innerText = asc ? min : max;
	if (elem.textContent)
		elem.textContent = asc ? min : max;

	elem = document.getElementById(divId + 'E');
	if (elem.innerText) 
		elem.innerText = asc ? max : min;
	if (elem.textContent)
		elem.textContent = asc ? max : min;
}

function measuresHide() {

	var showSelect = document.getElementById('measures-show');
	var showOptions = showSelect.options;

	while (true) {
		var found = false;
		for (var n = 0; n &lt; showOptions.length; n++) {
			if (showOptions[n].selected) {
				showSelect.remove(n);
				found = true;
			}
		}
		if (!found)
			break;
	}

	if (showOptions.length &gt; 0)
		showOptions[0].selected = true;
}

function measuresShow() {

	var allOptions = document.getElementById('measures-all').options;
	var showOptions = document.getElementById('measures-show').options;

	for (var n = 0; n &lt; allOptions.length; n++) {
		if (allOptions[n].selected)
			measuresShowAdd(showOptions, allOptions[n]);
	}
}

function measuresShowAdd(showOptions, option) {

	for (var n = 0; n &lt; showOptions.length; n++) {
		if (showOptions[n].value == option.value) {
			alert(option.text + ' is already in the "Measures Shown" box.');
			return;
		}
	}

	var newoption = new Option(option.text, option.value);
	showOptions[showOptions.length] = newoption;

}

function measuresSortDown() {

	var showSelect = document.getElementById('measures-show');
	var showOptions = showSelect.options;
	for (var n = showOptions.length - 2; n &gt; -1; n--) {
		if (!showOptions[n].selected)
			continue;
		// just swapping elements caused array to shorten by 1, need to make copy
		var nextCopy = new Option(showOptions[n+1].text, showOptions[n+1].value);
		var cur = showOptions[n];
		showOptions[n] = nextCopy;
		showOptions[n + 1] = cur;
	}
}

function measuresSortUp() {

	var showSelect = document.getElementById('measures-show');
	var showOptions = showSelect.options;
	for (var n = 1; n &lt; showOptions.length; n++) {
		if (!showOptions[n].selected)
			continue;
		// just swapping elements caused array to shorten by 1, need to make copy
		var prevCopy = new Option(showOptions[n-1].text, showOptions[n-1].value);
		var cur = showOptions[n];
		showOptions[n] = prevCopy;
		showOptions[n - 1] = cur;
	}
}

function setFocus(elemId) {

	var elem = document.getElementById(elemId);
	if (elem != null) {
		if (elemId = 'table-options')
			optionsShow('results_table');
		elem.scrollIntoView();
		window.scrollBy(0, -80);
	}
}

function validate() {

	// get measures from the Measures Shown list
	var showOptions = document.getElementById('measures-show').options;
	var measures = '';

	for (var n = 0; n &lt; showOptions.length; n++) {
		if (measures != '')
			measures += ',';
		measures += showOptions[n].value;
	}

	// if Change Table button clicked, perform validations, return false if any fail
	// otherwise set O_results_table_options.value = 'off' so table isn't redisplayed and return true
	if (submitGet() == 'action-Change Table') {
		// get measure selected in the Sort By list
		var sortSelect = document.getElementById('tsort');
		var index = sortSelect.selectedIndex;
		if (index == -1) {
			alert('Please select an item in the "Sort By" list');
			return false;
		}
		var sortOptions = sortSelect.options;
		var sel = sortOptions[index].value;
		var mcode = sel.split(',')[0];

		// check that sort measure is also shown
		if (mcode != 'byvar' &amp;&amp; measures.indexOf(mcode) == -1) {
			alert(sortOptions[index].text + '  is selected in the "Sort By" list but isn\'t in the "Measures Shown" list.'
				+ '\n\nPlease alter your selections so the sort by measure is also shown.');
			return false;
		}

		// if doing group breaks, make sure we're sorting by a measure
//		if (document.form.O_groups.value == 'true') {
		if (document.form.O_groups.checked) {
			// points to an element containing start value for numeric measures, isn't there for non-numeric
			var elem = document.getElementById(mcode + '_MEASURE_ASCENDS');
			if (mcode == 'byvar' || elem == null) {
				alert('A numeric measure must be selected in the "Sort By" list to break results into groups.');
				return false;
			}
		}

		document.form.O_results_table_options.value = 'off';
	}
	

	// measures_list contains the complete list from the Measures Shown list, needed by Java code
	// normally just selected items returned cause it's a select box
	document.form.measures_list.value = measures;

	return sendMsgSet();
}
